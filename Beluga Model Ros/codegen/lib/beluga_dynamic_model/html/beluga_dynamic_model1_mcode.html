1function newstate = beluga_dynamic_model( state, u, disturbance, Ts )
 2%SURFACE_VEHICLE_DYNAMIC_MODEL 
 3% Author: Vaibhav Viswanathan (vviswanathan@hmc.edu)
 4%             (email me for any questions!)
 5%                    ___
 6%  current state -> |   |
 7%                   | H | -> new state
 8% control inputs -> |___|
 9
10integration_steps = 5; %increase this number if you find the 
11                       %simulator go into numerical instability
12
13%% CONSTANTS TODO UPDATE
14m = 1; % mass 1kg
15I = 1;
16k1 = 10; % 1/2*p*Cd*A; drag coefficient
17k2 = 5; % rotational drag
18r = 0.050; % thrusters r meters from the center of mass axis
19a = 0.050; % center of drag a meters from center of mass along axis
20max_thrust = 500; % Newtons
21
22mass_matrix = ones(6);
23
24%% STATE
25x       = state(1);
26y       = state(2);
27z       = state(3);
28phi     = state(4);
29theta   = state(5);
30psi     = state(6);
31
32x_d   = state(1);
33y_d   = state(2);
34z_d   = state(3);
35phi_d = state(4);
36theta_d = state(5);
37psi_d = state(6);
38
39x_disturbance     = disturbance(1);
40y_disturbance     = disturbance(2);
41theta_disturbance = disturbance(3);
42
43%% CALCULATE FORCES TODO
44
45Fl = max_thrust*u(1);
46Fr = max_thrust*u(2);
47
48F = ones(6,1); 
49
50%% COMPUTE NEW STATE TODO
51
52% Convert forces into accelerations
53accel = (mass_matrix)\F;
54state_d = [state(1:6), accel'];
55
56newstate = state(1:12);
57
58for j=1:integration_steps
59    
60    Ts_int = Ts/integration_steps;
61
62    % update state variables by integrating
63    newstate = newstate + state_d*Ts_int;
64
65end
66% convert to global frame
67
68newstate = [newstate, u(1), u(2), u(3), u(4)];
69
70end
